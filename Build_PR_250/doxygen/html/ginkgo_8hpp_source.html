46 auto valuesView =
47 gko::array<scalar>::view(gkoExec_, mtx.values().size(), mtx.values().data());
48 auto colIdxView =
49 gko::array<int>::view(gkoExec_, mtx.colIdxs().size(), mtx.colIdxs().data());
50 auto rowPtrView =
51 gko::array<int>::view(gkoExec_, mtx.rowPtrs().size(), mtx.rowPtrs().data());
52 size_t nrows = sys.rhs().size();
53
54 auto maxIters = size_t(solverDict_.get<int>("maxIters"));
55 auto relTol = solverDict_.get<float>("relTol");
56 auto solverFact =
57 gko::solver::Cg<ValueType>::build()
58 .with_criteria(
59 gko::stop::Iteration::build().with_max_iters(maxIters),
60 gko::stop::ResidualNorm<ValueType>::build().with_reduction_factor(relTol)
61 )
62 .on(gkoExec_);
63
64 auto gkoA = gko::share(gko::matrix::Csr<scalar, int>::create(
65 gkoExec_, gko::dim<2> {nrows, nrows}, valuesView, colIdxView, rowPtrView
66 ));
67
68 auto solver = solverFact->generate(gkoA);
69
70 auto gkoRhs = gko::matrix::Dense<scalar>::create(
71 gkoExec_,
72 gko::dim<2> {nrows, 1},
73 gko::array<scalar>::view(gkoExec_, nrows, sys.rhs().data()),
74 1
75 );
76 auto gkoX = gko::matrix::Dense<scalar>::create(
77 gkoExec_, gko::dim<2> {nrows, 1}, gko::array<scalar>::view(gkoExec_, nrows, x.data()), 1
