68template<typename ValueType, typename IndexType>
69Field<ValueType> SpMV(LinearSystem<ValueType, IndexType>& ls, Field<ValueType>& xfield)
70{
71 Field<ValueType> resultField(ls.exec(), ls.rhs().size(), 0.0);
72 auto [result, b, x] = spans(resultField, ls.rhs(), xfield);
73
74 std::span<ValueType> values = ls.matrix().values();
75 std::span<IndexType> colIdxs = ls.matrix().colIdxs();
76 std::span<IndexType> rowPtrs = ls.matrix().rowPtrs();
77
78 NeoFOAM::parallelFor(
79 ls.exec(),
80 {0, ls.matrix().nRows()},
81 KOKKOS_LAMBDA(const std::size_t rowi) {
82 IndexType rowStart = rowPtrs[rowi];
83 IndexType rowEnd = rowPtrs[rowi + 1];
84 ValueType sum = 0.0;
85 for (IndexType coli = rowStart; coli < rowEnd; coli++)
86 {
87 sum += values[coli] * x[colIdxs[coli]];
88 }
89 result[rowi] = sum - b[rowi];
90 }
91 );
92
93 return resultField;
94};

97template<typename ValueType, typename SourceIndexType, typename DestIndexType = int>
98LinearSystem<ValueType, DestIndexType>
99convertLinearSystem(const LinearSystem<ValueType, SourceIndexType>& ls)
100{
101
102 Field<ValueType> convertedValues(
103 ls.exec(), ls.matrix().values().data(), ls.matrix().values().size()
104 );
105 Field<DestIndexType> convertedColIdxs(ls.exec(), ls.matrix().colIdxs().size());
106 auto mColIdxsSpan = ls.matrix().colIdxs();
107 NeoFOAM::parallelFor(
108 convertedColIdxs, KOKKOS_LAMBDA(const size_t i) { return DestIndexType(mColIdxsSpan[i]); }
109 );
110
111 Field<DestIndexType> convertedRowPtrs(ls.exec(), ls.matrix().rowPtrs().size());
112
113 auto mRowPtrsSpan = ls.matrix().rowPtrs();
114 NeoFOAM::parallelFor(
115 convertedRowPtrs, KOKKOS_LAMBDA(const size_t i) { return DestIndexType(mRowPtrsSpan[i]); }
116 );
117
118 Field<ValueType> convertedRhs(ls.exec(), ls.rhs().data(), ls.rhs().size());
119
120 la::CSRMatrix<ValueType, DestIndexType> matrix(
121 convertedValues, convertedColIdxs, convertedRowPtrs
122 );
123
124 return {matrix, convertedRhs, ls.sparsityPattern()};
125}
