31 void explicitOperation(Field<scalar>& source, scalar t, scalar dt)

33 const scalar rDeltat = 1 / dt;
34 const auto vol = getField().mesh().cellVolumes().span();
35 auto operatorScaling = getCoefficient();
36 auto [sourceSpan, field, oldField] =

40 source.range(),
41 KOKKOS_LAMBDA(const size_t celli) {

52 void implicitOperation(la::LinearSystem<scalar, localIdx>& ls, scalar t, scalar dt)
53 {
54 const scalar rDeltat = 1 / dt;
55 const auto vol = getField().mesh().cellVolumes().span();
56 const auto operatorScaling = getCoefficient();
57 const auto [diagOffs, oldField] =
58 spans(sparsityPattern_->diagOffset(), oldTime(field_).internalField());
59 const auto rowPtrs = ls.matrix().rowPtrs();
60 const auto colIdxs = ls.matrix().colIdxs();
61 std::span<scalar> values = ls.matrix().values();
62 std::span<scalar> rhs = ls.rhs().span();
63 NeoFOAM::parallelFor(
64 ls.exec(),
65 {0, oldField.size()},
66 KOKKOS_LAMBDA(const size_t celli) {
67 std::size_t idx = rowPtrs[celli] + diagOffs[celli];
68 values[idx] += operatorScaling[celli] * rDeltat * vol[celli];
69 rhs[celli] += operatorScaling[celli] * rDeltat * oldField[celli] * vol[celli];
70 }
71 );
72 }
