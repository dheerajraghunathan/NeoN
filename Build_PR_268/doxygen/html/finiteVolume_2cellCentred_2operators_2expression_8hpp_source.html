20template<typename ValueType, typename IndexType = localIdx>
21la::LinearSystem<ValueType, IndexType> convert(const la::LinearSystem<scalar, IndexType>& ls)
22{
23
24 Field<ValueType> values(ls.matrix().exec(), ls.matrix().nNonZeros(), zero<ValueType>());
25 Field<localIdx> mColIdxs(
26 ls.matrix().exec(), ls.matrix().colIdxs().data(), ls.matrix().nNonZeros()
27 );
28 Field<localIdx> mRowPtrs(
29 ls.matrix().exec(), ls.matrix().rowPtrs().data(), ls.matrix().rowPtrs().size()
30 );
31
32 la::CSRMatrix<ValueType, localIdx> matrix(values, mColIdxs, mRowPtrs);
33 Field<ValueType> rhs(ls.matrix().exec(), ls.rhs().size(), zero<ValueType>());
34
35 return {matrix, rhs, ls.sparsityPattern()};
36}

89 void assemble(scalar t, scalar dt)
90 {
91 auto vol = psi_.mesh().cellVolumes().span();
92 auto expSource = expr_.explicitOperation(psi_.mesh().nCells());
93 expr_.explicitOperation(expSource, t, dt);
94 auto expSourceSpan = expSource.span();
95
96 ls_ = expr_.implicitOperation();

99 // we subtract the explicit source term from the rhs

101 exec(),
102 {0, rhs.size()},
103 KOKKOS_LAMBDA(const size_t i) { rhs[i] -= expSourceSpan[i] * vol[i]; }

166 // TODO currently assumes that matrix is already assembled

168 const auto rowPtrs = ls_.matrix().rowPtrs();
169 auto rhs = ls_.rhs().span();

172 ls_.exec(),

175 auto diagIdx = rowPtrs[refCelli] + diagOffset[refCelli];
176 auto diagValue = values[diagIdx];
177 rhs[refCelli] += diagValue * refValue;

183 Field<ValueType> flux() const
184 {
185 const UnstructuredMesh& mesh = psi_.mesh();
186 const std::size_t nInternalFaces = mesh.nInternalFaces();
187 const auto exec = psi_.exec();
188 const auto [owner, neighbour, surfFaceCells, ownOffs, neiOffs, internalPsi] = spans(
189 mesh.faceOwner(),
190 mesh.faceNeighbour(),
191 mesh.boundaryMesh().faceCells(),
192 sparsityPattern_->ownerOffset(),
193 sparsityPattern_->neighbourOffset(),
194 psi_.internalField()
195 );
196
197 const auto rowPtrs = ls_.matrix().rowPtrs();
198 const auto colIdxs = ls_.matrix().colIdxs();
199 auto values = ls_.matrix().values();
200 auto rhs = ls_.rhs().span();
201
202
203 Field<ValueType> result(exec, neighbour.size(), 0.0);
204
205 auto resultSpan = result.span();
206
207 parallelFor(
208 exec,
209 {0, nInternalFaces},
210 KOKKOS_LAMBDA(const size_t facei) {
211 std::size_t own = static_cast<std::size_t>(owner[facei]);
212 std::size_t nei = static_cast<std::size_t>(neighbour[facei]);
213
214 std::size_t rowNeiStart = rowPtrs[nei];
215 std::size_t rowOwnStart = rowPtrs[own];
216
217 auto Upper = values[rowNeiStart + neiOffs[facei]];
218 auto Lower = values[rowOwnStart + ownOffs[facei]];
219 Kokkos::atomic_add(
220 &resultSpan[facei], Upper * internalPsi[nei] - Lower * internalPsi[own]
221 );
222 }
223 );
224 return result;
225 }

237template<typename ValueType, typename IndexType = localIdx>
238VolumeField<ValueType>
239operator&(const Expression<ValueType, IndexType> ls, const VolumeField<ValueType>& psi)
240{
241 VolumeField<ValueType> resultField(
242 psi.exec(),
243 "ls_" + psi.name,
244 psi.mesh(),
245 psi.internalField(),
246 psi.boundaryField(),
247 psi.boundaryConditions()
248 );
249
250 auto [result, b, x] =
251 spans(resultField.internalField(), ls.linearSystem().rhs(), psi.internalField());
252
253 const std::span<const ValueType> values = ls.linearSystem().matrix().values();
254 const std::span<const IndexType> colIdxs = ls.linearSystem().matrix().colIdxs();
255 const std::span<const IndexType> rowPtrs = ls.linearSystem().matrix().rowPtrs();
256
257 NeoFOAM::parallelFor(
258 resultField.exec(),
259 {0, result.size()},
260 KOKKOS_LAMBDA(const std::size_t rowi) {
261 IndexType rowStart = rowPtrs[rowi];
262 IndexType rowEnd = rowPtrs[rowi + 1];
263 ValueType sum = 0.0;
264 for (IndexType coli = rowStart; coli < rowEnd; coli++)
265 {
266 sum += values[coli] * x[colIdxs[coli]];
267 }
268 result[rowi] = sum - b[rowi];
269 }
270 );
271
272 return resultField;
273}
