16template<typename ValueType>
17void surfaceIntegrate(
18 const Executor& exec,
19 size_t nInternalFaces,
20 std::span<const int> neighbour,
21 std::span<const int> owner,
22 std::span<const int> faceCells,
23 std::span<const ValueType> flux,
24 std::span<const scalar> V,
25 std::span<ValueType> res,
26 const dsl::Coeff operatorScaling
27)
28{
29 size_t nCells {V.size()};
30 const size_t nBoundaryFaces = faceCells.size();
31 // check if the executor is GPU
32 if (std::holds_alternative<SerialExecutor>(exec))
33 {
34 for (size_t i = 0; i < nInternalFaces; i++)
35 {
36 res[static_cast<size_t>(owner[i])] += flux[i];
37 res[static_cast<size_t>(neighbour[i])] -= flux[i];
38 }
39
40 for (size_t i = nInternalFaces; i < nInternalFaces + nBoundaryFaces; i++)
41 {
42 auto own = static_cast<size_t>(faceCells[i - nInternalFaces]);
43 res[own] += flux[i];
44 }
45
46 // TODO does it make sense to store invVol and multiply?
47 for (size_t celli = 0; celli < nCells; celli++)
48 {
49 res[celli] *= operatorScaling[celli] / V[celli];
50 }
51 }
52 else
53 {
54 parallelFor(
55 exec,
56 {0, nInternalFaces},
57 KOKKOS_LAMBDA(const size_t i) {
58 Kokkos::atomic_add(&res[static_cast<size_t>(owner[i])], flux[i]);
59 Kokkos::atomic_sub(&res[static_cast<size_t>(neighbour[i])], flux[i]);
60 }
61 );
62
63 parallelFor(
64 exec,
65 {nInternalFaces, nInternalFaces + nBoundaryFaces},
66 KOKKOS_LAMBDA(const size_t i) {
67 auto own = static_cast<size_t>(faceCells[i - nInternalFaces]);
68 Kokkos::atomic_add(&res[own], flux[i]);
69 }
70 );
71
72 parallelFor(
73 exec,
74 {0, nCells},
75 KOKKOS_LAMBDA(const size_t celli) { res[celli] *= operatorScaling[celli] / V[celli]; }
76 );
77 }
78}
