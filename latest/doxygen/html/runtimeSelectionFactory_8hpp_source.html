305 static bool addSubType()
306 {
307 CreatorFunc func = [](Args... args) -> std::unique_ptr<Base>
308 { return std::unique_ptr<Base>(new derivedClass(std::forward<Args>(args)...)); };
309 RuntimeSelectionFactory::table()[derivedClass::name()] = func;
310
311 DerivedClassDocumentation childData;
312 childData.doc = []() -> std::string { return derivedClass::doc(); };
313 childData.schema = []() -> std::string { return derivedClass::schema(); };
314 RuntimeSelectionFactory::docTable()[derivedClass::name()] = childData;
315
316 return true;
317 }
318
319 ~Register() override
320 {
321 if (REGISTERED)
322 {
323 const auto& tbl = RuntimeSelectionFactory::table();
324 const auto it = tbl.find(derivedClass::name());
325 REGISTERED = (it != tbl.end());
326 }
327 }
328 };

370 static void keyExistsOrError(const std::string& name)
371 {
372 const auto& tbl = table();
373 if (tbl.find(name) == tbl.end())
374 {
375 auto msg = std::string(" Could not find constructor for ") + name + "\n";
