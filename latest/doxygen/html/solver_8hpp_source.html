54template<typename FieldType>
55NeoFOAM::la::LinearSystem<typename FieldType::ElementType, int>
56ginkgoMatrix(Expression& exp, FieldType& solution)
57{
58 using ValueType = typename FieldType::ElementType;
59 auto vol = solution.mesh().cellVolumes().span();
60 auto expSource = exp.explicitOperation(solution.mesh().nCells());
61 auto expSourceSpan = expSource.span();
62
63 auto ls = exp.implicitOperation();
64 Field<ValueType> rhs(solution.exec(), ls.rhs().data(), ls.rhs().size());
65 auto rhsSpan = rhs.span();
66 // we subtract the explicit source term from the rhs
67 NeoFOAM::parallelFor(
68 solution.exec(),
69 {0, rhs.size()},
70 KOKKOS_LAMBDA(const size_t i) { rhsSpan[i] -= expSourceSpan[i] * vol[i]; }
71 );
72
73 Field<ValueType> mValues(
74 solution.exec(), ls.matrix().values().data(), ls.matrix().values().size()
75 );
76 Field<int> mColIdxs(solution.exec(), ls.matrix().colIdxs().size());
77 auto mColIdxsSpan = ls.matrix().colIdxs();
78 NeoFOAM::parallelFor(
79 mColIdxs, KOKKOS_LAMBDA(const size_t i) { return int(mColIdxsSpan[i]); }
80 );
81
82 Field<int> mRowPtrs(solution.exec(), ls.matrix().rowPtrs().size());
83 auto mRowPtrsSpan = ls.matrix().rowPtrs();
84 NeoFOAM::parallelFor(
85 mRowPtrs, KOKKOS_LAMBDA(const size_t i) { return int(mRowPtrsSpan[i]); }
86 );
87
88 auto values = ls.matrix().values();
89
90
91 la::CSRMatrix<ValueType, int> matrix(mValues, mColIdxs, mRowPtrs);
92
93 return {matrix, rhs, ls.sparsityPattern()};
94}

96/* @brief solve an expression
97 *
98 * @param exp - Expression which is to be solved/updated.
99 * @param solution - Solution field, where the solution will be 'written to'.
100 * @param t - the time at the start of the time step.
101 * @param dt - time step for the temporal integration
102 * @param fvSchemes - Dictionary containing spatial operator and time integration properties
103 * @param fvSolution - Dictionary containing linear solver properties
104 */
105template<typename FieldType>
106void solve(
107 Expression& exp,
108 FieldType& solution,
109 scalar t,
110 scalar dt,
111 [[maybe_unused]] const Dictionary& fvSchemes,
112 [[maybe_unused]] const Dictionary& fvSolution
113)
114{
115 // FIXME:
116 if (exp.temporalOperators().size() == 0 && exp.spatialOperators().size() == 0)
117 {
118 NF_ERROR_EXIT("No temporal or implicit terms to solve.");
119 }
120 exp.build(fvSchemes);
121 if (exp.temporalOperators().size() > 0)
122 {
123 // integrate equations in time
124 timeIntegration::TimeIntegration<FieldType> timeIntegrator(
125 fvSchemes.subDict("ddtSchemes"), fvSolution
126 );
127 timeIntegrator.solve(exp, solution, t, dt);
128 }
129 else
130 {
131 // solve sparse matrix system
132 using ValueType = typename FieldType::ElementType;
133 auto ls = ginkgoMatrix(exp, solution);
134
135
136 NeoFOAM::la::ginkgo::BiCGStab<ValueType> solver(solution.exec(), fvSolution);
137 solver.solve(ls, solution.internalField());
138 }
139}
