133 faceFlux_(faceFlux), divOperatorStrategy_(std::move(divOperatorStrategy)) {};
134
135 DivOperator(
136 dsl::Operator::Type termType,
137 const SurfaceField<scalar>& faceFlux,
138 VolumeField<ValueType>& phi
139 )
140 : dsl::OperatorMixin<VolumeField<scalar>>(phi.exec(), dsl::Coeff(1.0), phi, termType),
141 faceFlux_(faceFlux), divOperatorStrategy_(nullptr) {};
142
143
144 void explicitOperation(Field<scalar>& source)
145 {
146 if (divOperatorStrategy_ == nullptr)
147 {
148 NF_ERROR_EXIT("DivOperatorStrategy not initialized");
149 }
150 NeoFOAM::Field<NeoFOAM::scalar> tmpsource(source.exec(), source.size(), 0.0);
151 const auto operatorScaling = this->getCoefficient();
152 divOperatorStrategy_->div(tmpsource, faceFlux_, this->getField(), operatorScaling);
153 source += tmpsource;
154 }
155
156 void div(Field<ValueType>& divPhi)
157 {
158 const auto operatorScaling = this->getCoefficient();
159 divOperatorStrategy_->div(divPhi, faceFlux_, this->getField(), operatorScaling);
160 }
161
162 la::LinearSystem<scalar, localIdx> createEmptyLinearSystem() const
163 {
164 if (divOperatorStrategy_ == nullptr)
165 {
166 NF_ERROR_EXIT("DivOperatorStrategy not initialized");
167 }
168 return divOperatorStrategy_->createEmptyLinearSystem();
169 }
170
171 void implicitOperation(la::LinearSystem<scalar, localIdx>& ls)
172 {
173 if (divOperatorStrategy_ == nullptr)
174 {
175 NF_ERROR_EXIT("DivOperatorStrategy not initialized");
176 }
177 const auto operatorScaling = this->getCoefficient();
178 divOperatorStrategy_->div(ls, faceFlux_, this->getField(), operatorScaling);
179 }
180
181
182 void div(la::LinearSystem<scalar, localIdx>& ls)
183 {
184 const auto operatorScaling = this->getCoefficient();
185 divOperatorStrategy_->div(ls, faceFlux_, this->getField(), operatorScaling);
186 };
187
188 void div(VolumeField<scalar>& divPhi)
189 {
190 const auto operatorScaling = this->getCoefficient();
191 divOperatorStrategy_->div(divPhi, faceFlux_, this->getField(), operatorScaling);
192 }
193
194
195 void build(const Input& input)
196 {
197 const UnstructuredMesh& mesh = this->getField().mesh();
198 if (std::holds_alternative<NeoFOAM::Dictionary>(input))
199 {
200 auto dict = std::get<NeoFOAM::Dictionary>(input);
201 std::string schemeName = "div(" + faceFlux_.name + "," + this->getField().name + ")";
202 auto tokens = dict.subDict("divSchemes").get<NeoFOAM::TokenList>(schemeName);
203 divOperatorStrategy_ = DivOperatorFactory::create(this->exec(), mesh, tokens);
204 }
205 else
206 {
207 auto tokens = std::get<NeoFOAM::TokenList>(input);
208 divOperatorStrategy_ = DivOperatorFactory::create(this->exec(), mesh, tokens);
209 }
210 }
211
212 std::string getName() const { return "DivOperator"; }
213
214private:
215

217
218 std::unique_ptr<DivOperatorFactory> divOperatorStrategy_;
219};
