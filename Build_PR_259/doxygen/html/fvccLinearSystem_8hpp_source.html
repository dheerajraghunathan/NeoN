101VolumeField<ValueType>
102operator&(const LinearSystem<ValueType, IndexType> ls, const VolumeField<ValueType>& psi)
103{
104 VolumeField<ValueType> resultField(
105 psi.exec(),
106 "ls_" + psi.name,
107 psi.mesh(),
108 psi.internalField(),
109 psi.boundaryField(),
110 psi.boundaryConditions()
111 );
112
113 auto [result, b, x] =
114 spans(resultField.internalField(), ls.linearSystem().rhs(), psi.internalField());
115
116 const std::span<const ValueType> values = ls.linearSystem().matrix().values();
117 const std::span<const IndexType> colIdxs = ls.linearSystem().matrix().colIdxs();
118 const std::span<const IndexType> rowPtrs = ls.linearSystem().matrix().rowPtrs();
119
120 NeoFOAM::parallelFor(
121 resultField.exec(),
122 {0, result.size()},
123 KOKKOS_LAMBDA(const std::size_t rowi) {
124 IndexType rowStart = rowPtrs[rowi];
125 IndexType rowEnd = rowPtrs[rowi + 1];
126 ValueType sum = 0.0;
127 for (IndexType coli = rowStart; coli < rowEnd; coli++)
128 {
129 sum += values[coli] * x[colIdxs[coli]];
130 }
131 result[rowi] = sum - b[rowi];
132 }
133 );
134
135 return resultField;
136}
