226template<typename ValueTypeIn, typename IndexTypeIn, typename ValueTypeOut, typename IndexTypeOut>
227la::CSRMatrix<ValueTypeOut, IndexTypeOut>
228convert(const Executor exec, const la::CSRMatrixView<const ValueTypeIn, const IndexTypeIn> in)
229{
230 Field<IndexTypeOut> colIdxsTmp(exec, in.colIdxs.size());
231 Field<IndexTypeOut> rowPtrsTmp(exec, in.rowOffs.size());
232 Field<ValueTypeOut> valuesTmp(exec, in.values.data(), in.values.size());
233
234 parallelFor(
235 colIdxsTmp, KOKKOS_LAMBDA(const size_t i) { return IndexTypeOut(in.colIdxs[i]); }
236 );
237 parallelFor(
238 rowPtrsTmp, KOKKOS_LAMBDA(const size_t i) { return IndexTypeOut(in.rowOffs[i]); }
239 );
240 parallelFor(
241 valuesTmp, KOKKOS_LAMBDA(const size_t i) { return ValueTypeOut(in.values[i]); }
242 );
243
244 return la::CSRMatrix<ValueTypeOut, IndexTypeOut> {valuesTmp, colIdxsTmp, rowPtrsTmp};
245}
