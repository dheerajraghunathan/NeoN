109 DivOperator(

114 )

117
118 DivOperator(
119 dsl::Operator::Type termType,
120 const SurfaceField<scalar>& faceFlux,
121 VolumeField<ValueType>& phi

123 : dsl::OperatorMixin<VolumeField<scalar>>(phi.exec(), phi, termType), faceFlux_(faceFlux),
124 divOperatorStrategy_(nullptr) {};
125
126
127 void explicitOperation(Field<scalar>& source)
128 {
129 if (divOperatorStrategy_ == nullptr)
130 {
131 NF_ERROR_EXIT("DivOperatorStrategy not initialized");
132 }
133 NeoFOAM::Field<NeoFOAM::scalar> tmpsource(source.exec(), source.size(), 0.0);
134 divOperatorStrategy_->div(tmpsource, faceFlux_, this->getField());
135 source += tmpsource;
136 }
137
138 void div(Field<ValueType>& divPhi)
139 {
140 divOperatorStrategy_->div(divPhi, faceFlux_, this->getField());
141 }
142
143 la::LinearSystem<scalar, localIdx> createEmptyLinearSystem() const
144 {
145 if (divOperatorStrategy_ == nullptr)
146 {
147 NF_ERROR_EXIT("DivOperatorStrategy not initialized");
148 }
149 return divOperatorStrategy_->createEmptyLinearSystem();
150 }
151
152 void implicitOperation(la::LinearSystem<scalar, localIdx>& ls)
153 {
154 if (divOperatorStrategy_ == nullptr)
155 {
156 NF_ERROR_EXIT("DivOperatorStrategy not initialized");
157 }
158 divOperatorStrategy_->div(ls, faceFlux_, this->getField());
159 }
160
161
162 void div(la::LinearSystem<scalar, localIdx>& ls)
163 {
164 divOperatorStrategy_->div(ls, faceFlux_, this->getField());
165 };
166
167 void div(VolumeField<scalar>& divPhi)
168 {
169 divOperatorStrategy_->div(divPhi, faceFlux_, this->getField());
170 }
171
172
173 void build(const Input& input)
174 {
175 const UnstructuredMesh& mesh = this->getField().mesh();
176 if (std::holds_alternative<NeoFOAM::Dictionary>(input))
177 {
178 auto dict = std::get<NeoFOAM::Dictionary>(input);
179 std::string schemeName = "div(" + faceFlux_.name + "," + this->getField().name + ")";
180 auto tokens = dict.subDict("divSchemes").get<NeoFOAM::TokenList>(schemeName);
181 divOperatorStrategy_ = DivOperatorFactory::create(this->exec(), mesh, tokens);
182 }
183 else
184 {
185 auto tokens = std::get<NeoFOAM::TokenList>(input);
186 divOperatorStrategy_ = DivOperatorFactory::create(this->exec(), mesh, tokens);
187 }
188 }
189
190 std::string getName() const { return "DivOperator"; }
191
192private:
193
194 const SurfaceField<NeoFOAM::scalar>& faceFlux_;
195
196 std::unique_ptr<DivOperatorFactory> divOperatorStrategy_;
197};
