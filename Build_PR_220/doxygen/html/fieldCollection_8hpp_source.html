345template<typename FieldType>
346class CreateFromExistingField
347{
348public:
349
350 std::string name;
351 const FieldType& field;
352 std::int64_t timeIndex = std::numeric_limits<std::int64_t>::max();
353 std::int64_t iterationIndex = std::numeric_limits<std::int64_t>::max();
354 std::int64_t subCycleIndex = std::numeric_limits<std::int64_t>::max();
355
356 FieldDocument operator()(Database& db)
357 {
358 VolumeField<scalar> vf(
359 field.exec(), name, field.mesh(), field.boundaryConditions(), db, "", ""
360 );
361
362 if (field.registered())
363 {
364 const FieldCollection& fieldCollection = FieldCollection::instance(field);
365 const FieldDocument& fieldDoc = fieldCollection.fieldDoc(field.key);
366 if (timeIndex == std::numeric_limits<std::int64_t>::max())
367 {
368 timeIndex = fieldDoc.timeIndex();
369 }
370 if (iterationIndex == std::numeric_limits<std::int64_t>::max())
371 {
372 iterationIndex = fieldDoc.iterationIndex();
373 }
374 if (subCycleIndex == std::numeric_limits<std::int64_t>::max())
375 {
376 subCycleIndex = fieldDoc.subCycleIndex();
377 }
378 }
379 return NeoFOAM::Document(
380 {{"name", vf.name},
381 {"timeIndex", timeIndex},
382 {"iterationIndex", iterationIndex},
383 {"subCycleIndex", subCycleIndex},
384 {"field", vf}},
385 validateFieldDoc
386 );
387 }
388};
