23template<typename IndexType>
24IndexType segmentsFromIntervals(const Field<IndexType>& intervals, Field<IndexType>& offsets)
25{
26 IndexType finalValue = 0;
27 const auto inSpan = intervals.span();
28 // skip the first element of the offsets
29 // assumed to be zero
30 auto offsSpan = offsets.span().subspan(1);
31 NF_ASSERT_EQUAL(inSpan.size(), offsSpan.size());
32 NeoN::parallelScan(
33 intervals.exec(),
34 {0, offsSpan.size()},
35 KOKKOS_LAMBDA(const std::size_t i, IndexType& update, const bool final) {
36 update += inSpan[i];
37 if (final)
38 {
39 offsSpan[i] = update;
40 }
41 },
42 finalValue
43 );
44 return finalValue;
45}
