34 const scalar dtInver = 1.0 / dt;

41 source.range(),

50 const scalar dtInver = 1.0 / dt;
51 const auto vol = this->getField().mesh().cellVolumes().span();
52 const auto operatorScaling = this->getCoefficient();
53 const auto [diagOffs, oldField] =
54 spans(sparsityPattern_->diagOffset(), oldTime(this->field_).internalField());
55 auto [A, b] = ls.view();
56
57 NeoFOAM::parallelFor(
58 ls.exec(),
59 {0, oldField.size()},
60 KOKKOS_LAMBDA(const size_t celli) {
61 std::size_t idx = A.rowOffset[celli] + diagOffs[celli];
62 const auto commonCoef = operatorScaling[celli] * vol[celli] * dtInver;
63 A.value[idx] += commonCoef * one<ValueType>();
64 b[celli] += commonCoef * oldField[celli];
65 }
66 );
