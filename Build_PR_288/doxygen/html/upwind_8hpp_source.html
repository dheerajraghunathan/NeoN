19/* @brief computional kernel to perform an upwind interpolation
20** from a source volumeField to a surface field. It performs an interpolation
21** of the form
22**
23** d_f = w_f * s_O + ( 1 - w_f ) * s_N where w_1 is either 0,1 depending on the
24** direction of the flux
25**
26**@param src the input field
27**@param weights weights for the interpolation
28**@param dst the target field
29*/
30template<typename ValueType>
31void computeUpwindInterpolation(
32 const VolumeField<ValueType>& src,
33 const SurfaceField<scalar>& flux,
34 const SurfaceField<scalar>& weights,
35 SurfaceField<ValueType>& dst
36)
37{
38 const auto exec = dst.exec();
39 auto dstS = dst.internalField().span();
40 const auto [srcS, weightS, ownerS, neighS, boundS, fluxS] = spans(
41 src.internalField(),
42 weights.internalField(),
43 dst.mesh().faceOwner(),
44 dst.mesh().faceNeighbour(),
45 src.boundaryField().value(),
46 flux.internalField()
47 );
48 size_t nInternalFaces = dst.mesh().nInternalFaces();
49
50 NeoFOAM::parallelFor(
51 exec,
52 {0, dstS.size()},
53 KOKKOS_LAMBDA(const size_t facei) {
54 if (facei < nInternalFaces)
55 {
56 if (fluxS[facei] >= 0)
57 {
58 size_t own = static_cast<size_t>(ownerS[facei]);
59 dstS[facei] = srcS[own];
60 }
61 else
62 {
63 size_t nei = static_cast<size_t>(neighS[facei]);
64 dstS[facei] = srcS[nei];
65 }
66 }
67 else
68 {
69 dstS[facei] = weightS[facei] * boundS[facei - nInternalFaces];
70 }
71 },
72 "computeUpwindInterpolation"
73 );
74}
