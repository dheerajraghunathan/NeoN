20template<typename ValueType>
21void setFixedValue(
22 DomainField<ValueType>& domainField, std::pair<size_t, size_t> range, ValueType fixedValue
23)
24{
25 auto [refGradient, value, valueFraction, refValue] = spans(
26 domainField.boundaryField().refGrad(),
27 domainField.boundaryField().value(),
28 domainField.boundaryField().valueFraction(),
29 domainField.boundaryField().refValue()
30 );
31
32 NeoFOAM::parallelFor(
33 domainField.exec(),
34 range,
35 KOKKOS_LAMBDA(const size_t i) {
36 refValue[i] = fixedValue;
37 value[i] = fixedValue;
38 valueFraction[i] = 1.0; // only used refValue
39 refGradient[i] = fixedValue; // not used
40 }
41 );
42}
