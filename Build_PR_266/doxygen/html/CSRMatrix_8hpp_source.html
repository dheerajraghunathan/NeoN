40 KOKKOS_INLINE_FUNCTION
41 ValueType& entry(const IndexType i, const IndexType j) const
42 {
43 for (IndexType ic = 0; ic < (rowPtrs_[i + 1] - rowPtrs_[i]); ++ic)
44 {
45 if (colIdxs_[rowPtrs_[i] + ic] == j)
46 {
47 return values_[rowPtrs_[i] + ic];
48 }
49 if (colIdxs_[rowPtrs_[i] + ic] > j) break;
50 }
51 Kokkos::abort("Memory not allocated for CSR matrix component.");
