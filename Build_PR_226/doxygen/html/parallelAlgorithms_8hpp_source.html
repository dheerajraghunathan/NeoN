77// Concept to check if a callable is compatible with void(const size_t)
78template<typename Kernel>
79concept parallelForKernel = requires(Kernel t, size_t i) {
80 {
81 t(i)

86template<typename Executor, parallelForKernel Kernel>
87void parallelFor(
88 [[maybe_unused]] const Executor& exec,
89 std::pair<size_t, size_t> range,
90 Kernel kernel,
91 std::string name = "parallelFor"
92)
93{
94 auto [start, end] = range;
95 if constexpr (std::is_same<std::remove_reference_t<Executor>, SerialExecutor>::value)
96 {
97 for (size_t i = start; i < end; i++)
98 {
99 kernel(i);
100 }
101 }
102 else
103 {
104 using runOn = typename Executor::exec;
105 Kokkos::parallel_for(
106 name,
107 Kokkos::RangePolicy<runOn>(start, end),
108 KOKKOS_LAMBDA(const size_t i) { kernel(i); }
109 );
110 }
111}

191template<typename Executor, typename Kernel, typename T>
192void parallelReduce(
193 [[maybe_unused]] const Executor& exec, std::pair<size_t, size_t> range, Kernel kernel, T& value
194)
195{
196 auto [start, end] = range;
197 if constexpr (std::is_same<std::remove_reference_t<Executor>, SerialExecutor>::value)
198 {
199 for (size_t i = start; i < end; i++)
200 {
201 if constexpr (Kokkos::is_reducer<T>::value)
202 {
203 kernel(i, value.reference());
204 }
205 else
206 {
207 kernel(i, value);
208 }
209 }
210 }
211 else
212 {
213 using runOn = typename Executor::exec;
214 Kokkos::parallel_reduce(
215 "parallelReduce", Kokkos::RangePolicy<runOn>(start, end), kernel, value
216 );
217 }
218}
