71template<typename Executor, parallelForKernel Kernel>
72void parallelFor(
73 [[maybe_unused]] const Executor& exec,
74 std::pair<size_t, size_t> range,
75 Kernel kernel,
76 std::string name = "parallelFor"
77)
78{
79 auto [start, end] = range;
80 if constexpr (std::is_same<std::remove_reference_t<Executor>, SerialExecutor>::value)
81 {
82 for (size_t i = start; i < end; i++)
83 {
84 kernel(i);
85 }
86 }
87 else
88 {
89 using runOn = typename Executor::exec;
90 Kokkos::parallel_for(
91 name,
92 Kokkos::RangePolicy<runOn>(start, end),
93 KOKKOS_LAMBDA(const size_t i) { kernel(i); }
94 );
95 }
96}

172template<typename Executor, typename Kernel, typename T>
173void parallelReduce(
174 [[maybe_unused]] const Executor& exec, std::pair<size_t, size_t> range, Kernel kernel, T& value
175)
176{
177 auto [start, end] = range;
178 if constexpr (std::is_same<std::remove_reference_t<Executor>, SerialExecutor>::value)
179 {
180 for (size_t i = start; i < end; i++)
181 {
182 if constexpr (Kokkos::is_reducer<T>::value)
183 {
184 kernel(i, value.reference());
185 }
186 else
187 {
188 kernel(i, value);
189 }
190 }
191 }
192 else
193 {
194 using runOn = typename Executor::exec;
195 Kokkos::parallel_reduce(
196 "parallelReduce", Kokkos::RangePolicy<runOn>(start, end), kernel, value
197 );
198 }
199}
