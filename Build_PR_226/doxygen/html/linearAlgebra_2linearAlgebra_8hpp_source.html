50template<typename ValueType, typename IndexType>
51struct GinkgoMatrixAssembly
52{
53 GinkgoMatrixAssembly(Executor exec, size_t numRows, size_t numCols, size_t nnzEstimation)
54 : data(getGkoExecutor(exec), gko::dim<2>(numRows, numCols), nnzEstimation)
55 {
56 auto size = data.get_size();
57 auto arrays = data.empty_out();
58 arrays.row_idxs.fill(0);
59 arrays.col_idxs.fill(0);
60 arrays.values.fill(0);
61 data = gko::device_matrix_data<ValueType, IndexType>(
62 data.get_executor(),
63 size,
64 std::move(arrays.row_idxs),
65 std::move(arrays.col_idxs),
66 std::move(arrays.values)
67 );
68 rowIdxs = data.get_row_idxs();
69 colIdxs = data.get_col_idxs();
70 values = data.get_values();
71 }
72
73 KOKKOS_FUNCTION void insert(size_t nnzId, MatrixEntry<ValueType, IndexType> entry) const
74 {
75 rowIdxs[nnzId] = entry.row;
76 colIdxs[nnzId] = entry.col;
77 values[nnzId] = entry.value;
78 }
79
80 Executor exec() const { return exec_; }
81
82 auto&& getUnderlyingData() && { return std::move(data); }
83
84 auto& getUnderlyingData() & { return data; }
85
86 void finalize()
87 {
88 data.sort_row_major();
89 data.sum_duplicates();
90 }
91
92private:
93
94 Executor exec_;
95 gko::device_matrix_data<ValueType, label> data;
96
97 IndexType* rowIdxs;
98 IndexType* colIdxs;
99 ValueType* values;
100};

108 Matrix() : mtx(gko::matrix::Coo<ValueType, int>::create()) {}
109 Matrix(GinkgoMatrixAssembly<ValueType, int>&& matrixAssembly)
110 : mtx(gko::matrix::Coo<ValueType, int>::create(getGkoExecutor(matrixAssembly.exec())))
111 {
112 matrixAssembly.finalize();
113 auto gkoData = std::move(matrixAssembly).getUnderlyingData();
114
115 mtx->read(std::move(gkoData));
116 }
117
118 void apply(const Field<ValueType>& in, Field<ValueType>& out)
119 {
120 auto wrapField = [](auto& field)
121 {
122 auto data = const_cast<ValueType*>(field.data());
123 auto gkoExec = getGkoExecutor(field.exec());
124 return gko::matrix::Dense<ValueType>::create(
125 gkoExec,
126 gko::dim<2>(field.size(), 1),
127 gko::make_array_view(gkoExec, field.size(), data),
128 1
129 );
130 };
131 mtx->apply(wrapField(in), wrapField(out));
132 }
133
134 auto* getUnderlyingData() { return mtx.get(); }
135
136private:
137
138 std::shared_ptr<gko::matrix::Coo<ValueType, label>> mtx;
139};
