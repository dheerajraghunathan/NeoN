23template<typename IndexType>
24IndexType segmentsFromIntervals(const Vector<IndexType>& intervals, Vector<IndexType>& offsets)
25{
26 IndexType finalValue = 0;
27 const auto inSpan = intervals.view();
28 // skip the first element of the offsets
29 // assumed to be zero
30 auto offsSpan = offsets.view().subspan(1);
31 // NOTE avoid compiler warning by static_casting to localIdx since offsSpan
32 // is a std::span
33 NF_ASSERT_EQUAL(inSpan.size(), static_cast<localIdx>(offsSpan.size()));
34 NeoN::parallelScan(
35 intervals.exec(),
36 {0, offsSpan.size()},
37 KOKKOS_LAMBDA(const localIdx i, IndexType& update, const bool final) {
38 update += inSpan[i];
39 if (final)
40 {
41 offsSpan[i] = update;
42 }
43 },
44 finalValue
45 );
46 return finalValue;
47}
