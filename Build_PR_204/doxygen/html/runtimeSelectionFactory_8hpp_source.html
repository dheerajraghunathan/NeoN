310 static bool addSubType()
311 {
312 CreatorFunc func = [](Args... args) -> std::unique_ptr<Base>
313 { return std::unique_ptr<Base>(new derivedClass(std::forward<Args>(args)...)); };
314 RuntimeSelectionFactory::table()[derivedClass::name()] = func;
315
316 DerivedClassDocumentation childData;
317 childData.doc = []() -> std::string { return derivedClass::doc(); };
318 childData.schema = []() -> std::string { return derivedClass::schema(); };
319 RuntimeSelectionFactory::docTable()[derivedClass::name()] = childData;
320
321 return true;
322 }
323
324 ~Register() override
325 {
326 if (REGISTERED)
327 {
328 const auto& tbl = RuntimeSelectionFactory::table();
329 const auto it = tbl.find(derivedClass::name());
330 REGISTERED = (it != tbl.end());
331 }
332 }

379 static void keyExistsOrError(const std::string& name)

382 if (tbl.find(name) == tbl.end())

384 auto msg = std::string(" Could not find constructor for ") + name + "\n";
