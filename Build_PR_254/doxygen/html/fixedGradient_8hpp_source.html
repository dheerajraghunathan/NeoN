17// Without this function the compiler warns that calling a __host__ function
18// from a __device__ function is not allowed
19// NOTE: patchID was removed since it was unused
20// I guess it was replaced by range
21template<typename ValueType>
22void setGradientValue(
23 DomainField<ValueType>& domainField,
24 const UnstructuredMesh& mesh,
25 std::pair<size_t, size_t> range,
26 ValueType fixedGradient
27)
28{
29 const auto iField = domainField.internalField().span();
30
31 auto [refGradient, value, valueFraction, refValue] = spans(
32 domainField.boundaryField().refGrad(),
33 domainField.boundaryField().value(),
34 domainField.boundaryField().valueFraction(),
35 domainField.boundaryField().refValue()
36 );
37
38 auto faceCells = mesh.boundaryMesh().faceCells();
39 auto deltaCoeffs = mesh.boundaryMesh().deltaCoeffs();
40
41 NeoFOAM::parallelFor(
42 domainField.exec(),
43 range,
44 KOKKOS_LAMBDA(const size_t i) {
45 refGradient[i] = fixedGradient;
46 // operator / is not defined for all ValueTypes
47 value[i] =
48 iField[static_cast<size_t>(faceCells[i])] + fixedGradient * (1 / deltaCoeffs[i]);
49 valueFraction[i] = 0.0; // only used refGrad
50 refValue[i] = fixedGradient; // not used
51 },
52 "setGradientValue"
53 );
54}
55}
