20/* @brief computional kernel to perform a linear interpolation
21** from a source volumeField to a surface field. It performs an interpolation
22** of the form
23**
24** d_f = w_f * s_O + ( 1 - w_f ) * s_N
25**
26**@param src the input field
27**@param weights weights for the interpolation
28**@param dst the target field
29*/
30template<typename ValueType>
31void computeLinearInterpolation(
32 const VolumeField<ValueType>& src,
33 const SurfaceField<scalar>& weights,
34 SurfaceField<ValueType>& dst
35)
36{
37 const auto exec = dst.exec();
38 auto dstS = dst.internalField().span();
39 const auto [srcS, weightS, ownerS, neighS, boundS] = spans(
40 src.internalField(),
41 weights.internalField(),
42 dst.mesh().faceOwner(),
43 dst.mesh().faceNeighbour(),
44 src.boundaryField().value()
45 );
46 size_t nInternalFaces = dst.mesh().nInternalFaces();
47
48 NeoFOAM::parallelFor(
49 exec,
50 {0, dstS.size()},
51 KOKKOS_LAMBDA(const size_t facei) {
52 if (facei < nInternalFaces)
53 {
54 size_t own = static_cast<size_t>(ownerS[facei]);
55 size_t nei = static_cast<size_t>(neighS[facei]);
56 dstS[facei] = weightS[facei] * srcS[own] + (1 - weightS[facei]) * srcS[nei];
57 }
58 else
59 {
60 dstS[facei] = weightS[facei] * boundS[facei - nInternalFaces];
61 }
62 }
63 );
64}
