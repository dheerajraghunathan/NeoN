17/* @brief free standing function implementation of the divergence operator
18** ie computes 1/V \sum_f S_f \cdot \phi_f
19** where S_f is the face normal flux of a given face
20** phi_f is the face interpolate value
21**
22**
23** @param faceFlux
24** @param neighbour - mapping from face id to neighbour cell id
25** @param owner - mapping from face id to owner cell id
26** @param faceCells - mapping from boundary face id to owner cell id
27*/
28template<typename ValueType>
29void computeDiv(
30 const Executor& exec,
31 size_t nInternalFaces,
32 size_t nBoundaryFaces,
33 std::span<const int> neighbour,
34 std::span<const int> owner,
35 std::span<const int> faceCells,
36 std::span<const scalar> faceFlux,
37 std::span<const ValueType> phiF,
38 std::span<const scalar> V,
39 std::span<ValueType> res,
40 const dsl::Coeff operatorScaling
41)
42{
43 size_t nCells {V.size()};
44 // check if the executor is GPU
45 if (std::holds_alternative<SerialExecutor>(exec))
46 {
47 for (size_t i = 0; i < nInternalFaces; i++)
48 {
49 ValueType flux = faceFlux[i] * phiF[i];
50 res[static_cast<size_t>(owner[i])] += flux;
51 res[static_cast<size_t>(neighbour[i])] -= flux;
52 }
53
54 for (size_t i = nInternalFaces; i < nInternalFaces + nBoundaryFaces; i++)
55 {
56 auto own = static_cast<size_t>(faceCells[i - nInternalFaces]);
57 ValueType valueOwn = faceFlux[i] * phiF[i];
58 res[own] += valueOwn;
59 }
60
61 // TODO does it make sense to store invVol and multiply?
62 for (size_t celli = 0; celli < nCells; celli++)
63 {
64 res[celli] *= operatorScaling[celli] / V[celli];
65 }
66 }
67 else
68 {
69 parallelFor(
70 exec,
71 {0, nInternalFaces},
72 KOKKOS_LAMBDA(const size_t i) {
73 ValueType flux = faceFlux[i] * phiF[i];
74 Kokkos::atomic_add(&res[static_cast<size_t>(owner[i])], flux);
75 Kokkos::atomic_sub(&res[static_cast<size_t>(neighbour[i])], flux);
76 },
77 "sumFluxesInternal"
78 );
79
80 parallelFor(
81 exec,
82 {nInternalFaces, nInternalFaces + nBoundaryFaces},
83 KOKKOS_LAMBDA(const size_t i) {
84 auto own = static_cast<size_t>(faceCells[i - nInternalFaces]);
85 ValueType valueOwn = faceFlux[i] * phiF[i];
86 Kokkos::atomic_add(&res[own], valueOwn);
87 },
88 "sumFluxesInternal"
89 );
90
91 parallelFor(
92 exec,
93 {0, nCells},
94 KOKKOS_LAMBDA(const size_t celli) { res[celli] *= operatorScaling[celli] / V[celli]; },
95 "normalizeFluxes"
96 );
97 }
98}

101template<typename ValueType>
102void computeDiv(
103 const SurfaceField<scalar>& faceFlux,
104 const VolumeField<ValueType>& phi,
105 const SurfaceInterpolation<ValueType>& surfInterp,
106 Field<ValueType>& divPhi,
107 const dsl::Coeff operatorScaling
108)
109{
110 const UnstructuredMesh& mesh = phi.mesh();
111 const auto exec = phi.exec();
112 SurfaceField<ValueType> phif(
113 exec, "phif", mesh, createCalculatedBCs<SurfaceBoundary<ValueType>>(mesh)
114 );
115 // TODO: remove or implement
116 // fill(phif.internalField(), NeoFOAM::zero<ValueType>::value);
117 surfInterp.interpolate(faceFlux, phi, phif);
118
119 // TODO: currently we just copy the boundary values over
120 phif.boundaryField().value() = phi.boundaryField().value();
121
122 size_t nInternalFaces = mesh.nInternalFaces();
123 size_t nBoundaryFaces = mesh.nBoundaryFaces();
124 computeDiv<ValueType>(
125 exec,
126 nInternalFaces,
127 nBoundaryFaces,
128 mesh.faceNeighbour().span(),
129 mesh.faceOwner().span(),
130 mesh.boundaryMesh().faceCells().span(),
131 faceFlux.internalField().span(),
132 phif.internalField().span(),
133 mesh.cellVolumes().span(),
134 divPhi.span(),
135 operatorScaling
136
137 );
138}

140template<typename ValueType>
141class GaussGreenDiv :
142 public DivOperatorFactory<ValueType>::template Register<GaussGreenDiv<ValueType>>
143{
144 using Base = DivOperatorFactory<ValueType>::template Register<GaussGreenDiv<ValueType>>;
145
146public:
147
148 static std::string name() { return "Gauss"; }
149
150 static std::string doc() { return "Gauss-Green Divergence"; }
151
152 static std::string schema() { return "none"; }
153
154 GaussGreenDiv(const Executor& exec, const UnstructuredMesh& mesh, const Input& inputs)
155 : Base(exec, mesh), surfaceInterpolation_(exec, mesh, inputs),
156 sparsityPattern_(SparsityPattern::readOrCreate(mesh)) {};
157
158 la::LinearSystem<ValueType, localIdx> createEmptyLinearSystem() const override
159 {
160 la::LinearSystem<scalar, localIdx> ls(sparsityPattern_->linearSystem());
161 auto [A, b] = ls.view();
162 const auto& exec = ls.exec();
163
164 Field<ValueType> values(exec, A.value.size(), zero<ValueType>());
165 Field<localIdx> mColIdxs(exec, A.columnIndex.data(), A.columnIndex.size());
166 Field<localIdx> mRowPtrs(exec, A.rowOffset.data(), A.rowOffset.size());
167
168 la::CSRMatrix<ValueType, localIdx> matrix(values, mColIdxs, mRowPtrs);
169 Field<ValueType> rhs(exec, b.size(), zero<ValueType>());
170
171 return {matrix, rhs, ls.sparsityPattern()};
172 };

174 virtual void
175 div(VolumeField<ValueType>& divPhi,
176 const SurfaceField<scalar>& faceFlux,
177 const VolumeField<ValueType>& phi,
178 const dsl::Coeff operatorScaling) const override
179 {
180 computeDiv<ValueType>(
181 faceFlux, phi, surfaceInterpolation_, divPhi.internalField(), operatorScaling
182 );
183 }
184
185 virtual void
186 div(la::LinearSystem<ValueType, localIdx>& ls,
187 const SurfaceField<scalar>& faceFlux,
188 const VolumeField<ValueType>& phi,
189 const dsl::Coeff operatorScaling) const override
190 {
191 const UnstructuredMesh& mesh = phi.mesh();
192 const std::size_t nInternalFaces = mesh.nInternalFaces();
193 const auto exec = phi.exec();
194 const auto [sFaceFlux, owner, neighbour, diagOffs, ownOffs, neiOffs] = spans(
195 faceFlux.internalField(),
196 mesh.faceOwner(),
197 mesh.faceNeighbour(),
198 sparsityPattern_->diagOffset(),
199 sparsityPattern_->ownerOffset(),
200 sparsityPattern_->neighbourOffset()
201 );
202 auto [A, b] = ls.view();
203
204 parallelFor(
205 exec,
206 {0, nInternalFaces},
207 KOKKOS_LAMBDA(const size_t facei) {
208 scalar flux = sFaceFlux[facei];
209 // scalar weight = 0.5;
210 scalar weight = flux >= 0 ? 1 : 0;
211 ValueType value = zero<ValueType>();
212 std::size_t own = static_cast<std::size_t>(owner[facei]);
213 std::size_t nei = static_cast<std::size_t>(neighbour[facei]);
214
215 // add neighbour contribution upper
216 std::size_t rowNeiStart = A.rowOffset[nei];
217 std::size_t rowOwnStart = A.rowOffset[own];
218
219 value = -weight * flux * one<ValueType>();
220 // scalar valueNei = (1 - weight) * flux;
221 A.value[rowNeiStart + neiOffs[facei]] += value;
222 Kokkos::atomic_sub(&A.value[rowOwnStart + diagOffs[own]], value);
223
224 // upper triangular part
225 // add owner contribution lower
226 value = flux * (1 - weight) * one<ValueType>();
227 A.value[rowOwnStart + ownOffs[facei]] += value;
228 Kokkos::atomic_sub(&A.value[rowNeiStart + diagOffs[nei]], value);
229 }
230 );
231
232 parallelFor(
233 exec,
234 {0, b.size()},
235 KOKKOS_LAMBDA(const size_t celli) {
236 b[celli] *= operatorScaling[celli];
237 for (size_t i = A.rowOffset[celli]; i < A.rowOffset[celli + 1]; i++)
238 {
239 A.value[i] *= operatorScaling[celli];
240 }
241 }
242 );
243 };
