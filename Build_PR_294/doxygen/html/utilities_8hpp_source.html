37template<typename ValueType, typename IndexType>
38std::shared_ptr<gko::matrix::Csr<ValueType, int>>
39createGkoMtx(std::shared_ptr<const gko::Executor> exec, LinearSystem<ValueType, IndexType>& sys)
40{
41 size_t nrows = sys.rhs().size();
42 // TODO: avoid copying
43 // NOTE: since mtx is a converted copy we need to make sure that mtx data is not
44 // deallocated before solving
45 auto mtx = convert<ValueType, IndexType, ValueType, int>(sys.exec(), sys.view().matrix);
46 auto vals = createGkoArray(exec, mtx.values().span());
47 auto col = createGkoArray(exec, mtx.colIdxs().span());
48 auto row = createGkoArray(exec, mtx.rowPtrs().span());
49 return gko::share(
50 gko::matrix::Csr<ValueType, int>::create(exec, gko::dim<2> {nrows, nrows}, vals, col, row)
51 );
52}
