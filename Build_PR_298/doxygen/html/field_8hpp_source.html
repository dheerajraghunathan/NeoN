31auto deepCopyVisitor(size_t size, const ValueType* srcPtr, ValueType* dstPtr)
32{
33 return [size, srcPtr, dstPtr](const auto& srcExec, const auto& dstExec)
34 {
35 Kokkos::deep_copy(
36 dstExec.createKokkosView(dstPtr, size), srcExec.createKokkosView(srcPtr, size)
