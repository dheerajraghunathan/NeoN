21template<typename ValueType>
22void extrapolateValue(
23 DomainField<ValueType>& domainField,
24 const UnstructuredMesh& mesh,
25 std::pair<size_t, size_t> range
26)
27{
28 const auto iField = domainField.internalField().view();
29
30 auto [refGradient, value, valueFraction, refValue, faceCells] = spans(
31 domainField.boundaryField().refGrad(),
32 domainField.boundaryField().value(),
33 domainField.boundaryField().valueFraction(),
34 domainField.boundaryField().refValue(),
35 mesh.boundaryMesh().faceCells()
36 );
37
38
39 NeoFOAM::parallelFor(
40 domainField.exec(),
41 range,
42 KOKKOS_LAMBDA(const size_t i) {
43 // operator / is not defined for all ValueTypes
44 ValueType internalCellValue = iField[static_cast<size_t>(faceCells[i])];
45 value[i] = internalCellValue;
46 valueFraction[i] = 1.0; // only use refValue
47 refValue[i] = internalCellValue; // not used
48 refGradient[i] = zero<ValueType>();
49 },
50 "extrapolateValue"
51 );
52}
