100template<typename ValueType, typename IndexType>
101Field<ValueType> SpMV(LinearSystem<ValueType, IndexType>& ls, Field<ValueType>& xfield)
102{
103 Field<ValueType> resultField(ls.exec(), ls.rhs().size(), 0.0);
104 auto [result, b, x] = spans(resultField, ls.rhs(), xfield);
105
106 std::span<ValueType> values = ls.matrix().values();
107 std::span<IndexType> colIdxs = ls.matrix().colIdxs();
108 std::span<IndexType> rowPtrs = ls.matrix().rowPtrs();
109
110 NeoFOAM::parallelFor(
111 ls.exec(),
112 {0, ls.matrix().nRows()},
113 KOKKOS_LAMBDA(const std::size_t rowi) {
114 IndexType rowStart = rowPtrs[rowi];
115 IndexType rowEnd = rowPtrs[rowi + 1];
116 ValueType sum = 0.0;
117 for (IndexType coli = rowStart; coli < rowEnd; coli++)
118 {
119 sum += values[coli] * x[colIdxs[coli]];
120 }
121 result[rowi] = sum - b[rowi];
122 }
123 );
124
125 return resultField;
126};

129template<typename ValueType, typename SourceIndexType, typename DestIndexType = int>
130LinearSystem<ValueType, DestIndexType>
131convertLinearSystem(const LinearSystem<ValueType, SourceIndexType>& ls)
132{
133
134 Field<ValueType> convertedValues(
135 ls.exec(), ls.matrix().values().data(), ls.matrix().values().size()
136 );
137 Field<DestIndexType> convertedColIdxs(ls.exec(), ls.matrix().colIdxs().size());
138 auto mColIdxsSpan = ls.matrix().colIdxs();
139 NeoFOAM::parallelFor(
140 convertedColIdxs, KOKKOS_LAMBDA(const size_t i) { return DestIndexType(mColIdxsSpan[i]); }
141 );
142
143 Field<DestIndexType> convertedRowPtrs(ls.exec(), ls.matrix().rowPtrs().size());
144
145 auto mRowPtrsSpan = ls.matrix().rowPtrs();
146 NeoFOAM::parallelFor(
147 convertedRowPtrs, KOKKOS_LAMBDA(const size_t i) { return DestIndexType(mRowPtrsSpan[i]); }
148 );
149
150 Field<ValueType> convertedRhs(ls.exec(), ls.rhs().data(), ls.rhs().size());
151
152 la::CSRMatrix<ValueType, DestIndexType> matrix(
153 convertedValues, convertedColIdxs, convertedRowPtrs
154 );
155
156 return {matrix, convertedRhs, ls.sparsityPattern()};
157}
