141  void
142  solve(Expression& exp, SolutionFieldType& solutionField, scalar t, const scalar dt) override
143  {
144  // Setup sundials if required, load the current solution for temporal integration
145  if (PDEExpr_ == nullptr) initSUNERKSolver(exp, solutionField, t, dt);
146  NeoFOAM::sundials::fieldToNVector(solutionField.internalField(), solution_);
147  void* ark = reinterpret_cast<void*>(ODEMemory_.get());
148 
149  // Perform time integration
150  ARKodeSetFixedStep(ark, dt);
151  NeoFOAM::scalar timeOut;
152  auto stepReturn = ARKodeEvolve(ark, t + dt, solution_, &timeOut, ARK_ONE_STEP);
153 
154  // Post step checks
155  NF_ASSERT_EQUAL(stepReturn, 0);
156  NF_ASSERT_EQUAL(t + dt, timeOut);
157 
158  // Copy solution out. (Fence is in sundails free)
159  NeoFOAM::sundials::NVectorToField(solution_, solutionField.internalField());
160  }
