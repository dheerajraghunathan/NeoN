34 template<typename SolutionFieldType>
35 int explicitSolveWrapperFreeFunction(sunrealtype t, N_Vector y, N_Vector ydot, void* user_data)
36 {
37  // Pointer wrangling
38  NFData<SolutionFieldType>* nfData = reinterpret_cast<NFData<SolutionFieldType>*>(user_data);
39  sunrealtype* ydotarray = N_VGetArrayPointer(ydot);
40  sunrealtype* yarray = N_VGetArrayPointer(y);
41 
42  if (ydotarray == NULL)
43  {
44  return -1;
45  }
46  if (yarray == NULL)
47  {
48  return -1;
49  }
50 
51  // Copy initial value from y to source.
52  NeoFOAM::Field<NeoFOAM::scalar> source(nfData->system_->exec(), 1);
53  auto sourceData = source.data();
54  parallelFor(
55  source.exec(), source.range(), KOKKOS_LAMBDA(const size_t i) { sourceData[i] = yarray[i]; }
56  );
57 
58  source = nfData->system_->explicitOperation(source);
59  parallelFor(
60  source.exec(), source.range(), KOKKOS_LAMBDA(const size_t i) { ydotarray[i] = source[i]; }
61  );
62 
63  // check if execturo is GPU
64  if (std::holds_alternative<NeoFOAM::GPUExecutor>(nfData->system_->exec()))
65  {
66  Kokkos::fence();
67  }
68  return 0;
69 }
