65void fieldToNVectorImpl(const NeoFOAM::Field<ValueType>& field, N_Vector& vector)
66{
67 auto view = ::sundials::kokkos::GetVec<SKVectorType>(vector)->View();

163template<typename SolutionFieldType>
164int explicitRKSolve([[maybe_unused]] sunrealtype t, N_Vector y, N_Vector ydot, void* user_data)
165{
166 // Pointer wrangling
167 NeoFOAM::dsl::Expression* pdeExpre = reinterpret_cast<NeoFOAM::dsl::Expression*>(user_data);
168 sunrealtype* yDotArray = N_VGetArrayPointer(ydot);
169 sunrealtype* yArray = N_VGetArrayPointer(y);
170
171 NF_ASSERT(
172 yDotArray != nullptr && yArray != nullptr && pdeExpre != nullptr,
173 "Failed to dereference pointers in sundails."
174 );
175
176 // Copy initial value from y to source.
177 NeoFOAM::Field<NeoFOAM::scalar> source(pdeExpre->exec(), 1, 0.0);
178 source = pdeExpre->explicitOperation(source); // compute spatial
179 if (std::holds_alternative<NeoFOAM::GPUExecutor>(pdeExpre->exec()))
180 {
181 Kokkos::fence();
182 }
183 NeoFOAM::sundials::fieldToNVector(source, ydot); // assign rhs to ydot.
184 return 0;
185}
