188template<typename SolutionFieldType>
189int explicitRKSolve([[maybe_unused]] sunrealtype t, N_Vector y, N_Vector ydot, void* userData)
190{
191 // Pointer wrangling
192 NeoFOAM::dsl::Expression* pdeExpre = reinterpret_cast<NeoFOAM::dsl::Expression*>(userData);
193 sunrealtype* yDotArray = N_VGetArrayPointer(ydot);
194 sunrealtype* yArray = N_VGetArrayPointer(y);
195
196 NF_ASSERT(
197 yDotArray != nullptr && yArray != nullptr && pdeExpre != nullptr,
198 "Failed to dereference pointers in sundails."
199 );
200
201 size_t size = N_VGetLength(y);
202 // Copy initial value from y to source.
203 NeoFOAM::Field<NeoFOAM::scalar> source =
204 pdeExpre->explicitOperation(size) * -1.0; // compute spatial
205 if (std::holds_alternative<NeoFOAM::GPUExecutor>(pdeExpre->exec()))
206 {
207 Kokkos::fence();
208 }
209 NeoFOAM::sundials::fieldToSunNVector(source, ydot); // assign rhs to ydot.
210 return 0;
211}
