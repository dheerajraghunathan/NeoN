121template<typename SolutionFieldType>
122int explicitRKSolve([[maybe_unused]] sunrealtype t, N_Vector y, N_Vector ydot, void* user_data)
123{
124
125 // Pointer wrangling
126 NeoFOAM::dsl::Expression* pdeExpre = reinterpret_cast<NeoFOAM::dsl::Expression*>(user_data);
127 sunrealtype* yDotArray = N_VGetArrayPointer(ydot);
128 sunrealtype* yArray = N_VGetArrayPointer(y);
129
130 NF_ASSERT(
131 yDotArray != nullptr && yArray != nullptr && pdeExpre != nullptr,
132 "Failed to dereference pointers in sundails."
133 );
134
135 // Copy initial value from y to source.
136 NeoFOAM::Field<NeoFOAM::scalar> source(pdeExpre->exec(), 1, 0.0);
137 source = pdeExpre->explicitOperation(source); // compute spacial
138 if (std::holds_alternative<NeoFOAM::GPUExecutor>(pdeExpre->exec()))
139 {
140 Kokkos::fence();
141 }
142 NeoFOAM::sundials::fieldToNVector(source, ydot); // assign rhs to ydot.
143 return 0;
144}
