41ARKODE_ERKTableID stringToERKTable(const std::string& key)
42{
43 if (key == "Forward-Euler") return ARKODE_FORWARD_EULER_1_1;
44 if (key == "Heun")

174template<typename SolutionFieldType>
175int explicitRKSolve([[maybe_unused]] sunrealtype t, N_Vector y, N_Vector ydot, void* userData)
176{
177 // Pointer wrangling
178 NeoFOAM::dsl::Expression* pdeExpre = reinterpret_cast<NeoFOAM::dsl::Expression*>(userData);
179 sunrealtype* yDotArray = N_VGetArrayPointer(ydot);
180 sunrealtype* yArray = N_VGetArrayPointer(y);
181
182 NF_ASSERT(
183 yDotArray != nullptr && yArray != nullptr && pdeExpre != nullptr,
184 "Failed to dereference pointers in sundails."
185 );
186
187 size_t size = N_VGetLength(y);
188 // Copy initial value from y to source.
189 NeoFOAM::Field<NeoFOAM::scalar> source = pdeExpre->explicitOperation(size); // compute spatial
190 if (std::holds_alternative<NeoFOAM::GPUExecutor>(pdeExpre->exec()))
191 {
192 Kokkos::fence();
193 }
194 NeoFOAM::sundials::fieldToSunNVector(source, ydot); // assign rhs to ydot.
195 return 0;
196}
