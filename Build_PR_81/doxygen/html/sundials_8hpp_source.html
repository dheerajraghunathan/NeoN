41ARKODE_ERKTableID stringToERKTable(const std::string& key)
42{
43 if (key == "Forward Euler") return ARKODE_FORWARD_EULER_1_1;

171template<typename SolutionFieldType>
172int explicitRKSolve([[maybe_unused]] sunrealtype t, N_Vector y, N_Vector ydot, void* userData)
173{
174 // Pointer wrangling
175 NeoFOAM::dsl::Expression* pdeExpre = reinterpret_cast<NeoFOAM::dsl::Expression*>(userData);
176 sunrealtype* yDotArray = N_VGetArrayPointer(ydot);
177 sunrealtype* yArray = N_VGetArrayPointer(y);
178
179 NF_ASSERT(
180 yDotArray != nullptr && yArray != nullptr && pdeExpre != nullptr,
181 "Failed to dereference pointers in sundails."
182 );
183
184 // Copy initial value from y to source.
185 NeoFOAM::Field<NeoFOAM::scalar> source(pdeExpre->exec(), 1, 0.0);
186 source = pdeExpre->explicitOperation(source); // compute spatial
187 if (std::holds_alternative<NeoFOAM::GPUExecutor>(pdeExpre->exec()))
188 {
189 Kokkos::fence();
190 }
191 NeoFOAM::sundials::fieldToSunNVector(source, ydot); // assign rhs to ydot.
192 return 0;
193}
