65template<typename SKVectorType, typename ValueType>
66void fieldToNVectorImpl(const NeoFOAM::Field<ValueType>& field, N_Vector& vector)
67{
68 // Load the current solution for temporal integration
69 auto view = ::sundials::kokkos::GetVec<SKVectorType>(vector)->View();
70 NeoFOAM::parallelFor(
71 field.exec(), field.range(), KOKKOS_LAMBDA(const size_t i) { view(i) = field[i]; }
72 );
73};

166template<typename SolutionFieldType>
167int explicitRKSolve([[maybe_unused]] sunrealtype t, N_Vector y, N_Vector ydot, void* user_data)
168{
169 // Pointer wrangling
170 NeoFOAM::dsl::Expression* pdeExpre = reinterpret_cast<NeoFOAM::dsl::Expression*>(user_data);
171 sunrealtype* yDotArray = N_VGetArrayPointer(ydot);
172 sunrealtype* yArray = N_VGetArrayPointer(y);
173
174 NF_ASSERT(
175 yDotArray != nullptr && yArray != nullptr && pdeExpre != nullptr,
176 "Failed to dereference pointers in sundails."
177 );
178
179 // Copy initial value from y to source.
180 NeoFOAM::Field<NeoFOAM::scalar> source(pdeExpre->exec(), 1, 0.0);
181 source = pdeExpre->explicitOperation(source); // compute spacial
182 if (std::holds_alternative<NeoFOAM::GPUExecutor>(pdeExpre->exec()))
183 {
184 Kokkos::fence();
185 }
186 NeoFOAM::sundials::fieldToNVector(source, ydot); // assign rhs to ydot.
187 return 0;
188}

240template<typename ValueType>
241std::unique_ptr<SKVector> makeSKVector(const NeoFOAM::Executor& exec)
242{
243 if (std::holds_alternative<NeoFOAM::GPUExecutor>(exec))
244 {
245 return std::make_unique<SKVectorDefault<ValueType>>();
246 }
247 if (std::holds_alternative<NeoFOAM::CPUExecutor>(exec))
248 {
249 return std::make_unique<SKVectorDefault<ValueType>>();
250 }
251 if (std::holds_alternative<NeoFOAM::SerialExecutor>(exec))
252 {
253 return std::make_unique<SKVectorSerial<ValueType>>();
254 }
255 NF_ERROR_EXIT(
256 "Unsupported NeoFOAM executor "
257 << std::visit<Executor>([&](const auto& e) { return e.name(); }, exec) << "."
